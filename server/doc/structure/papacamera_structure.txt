----dbstructure        # SQL的库的初始化操作[建表]


handler  # 页面控制器
         # 注意，其中的BaseHandler封装了通用的一些操作，他继承自RequestHandler，其他的Handler都继承自BaseHandler
		 # base.py中的：BaseHandler完成了类似的操作：
		 #      获得各种module的内容
		 #      一些通用操作，比如模板位置的render
		 #      一些异常操作
	     # 注意：一个.py不需要像java一样，以一个类为单位，可以以一个功能点为单位，一个py文件写很多个相关的类
	     # 比如：可以写user.py里面包含LoginHandler,LogoutHandler...，而不是：login.py,logout.py,register.py





modules         #模型包--里面的类继承query类，封装了面向业务逻辑的接口。
					以业务逻辑为单位封装成一个个module，module包含了很多面向业务逻辑的对数据库的基本操作
					module的操作是不包含合法检测的，只是把参数按照方法定义，转化为sql或者mongo的语句
					这些module被加载成为BaseHandler的数据成员，方便其他的所有Handler直接操作
					create a modules depend on table

static        #静态资源
----css
----js
----pages
----images


upload        #用户上传的资源，tornado用户可以直接访问的只有static，其他的可以访问，你需要在application进行配置
				比如：
 			  #--这里要加入静态文件的直接访问。
    			handlers.append((r"/data/(.*)", tornado.web.StaticFileHandler,{"path":"data"}))#这是data静态文件，用自定义静态文件的路由。
			    handlers.append((r"/images/(.*)", tornado.web.StaticFileHandler,{"path":"images"}))#这是images静态文件，用自定义静态文件的路由。
			    handlers.append((r"/upload/(.*)", tornado.web.StaticFileHandler,{"path":"upload"}))#这是上传的静态文件，用自定义静态文件的路由。

---lib        #自定义库---用来处理一些复杂的内部逻辑的类的自定义的库，
				可以拿python自带的库再次封装修改而成，也可以继承自object，和自己的逻辑相关.其他的几个package引用自这个lib的内容
                    ---目的是根据自己的业务需求，实现一些基本操作的方法，并把他们封装起来
				比如：
					loader---封装了登录注册登出的方法
					jsonp----重新封装了原本的json，按需要的格式打印出来
					sendmail--- 封装了发送mail的方法
					query----封装了所有sql操作
			   lib.py can be an object or just a list of functions.
